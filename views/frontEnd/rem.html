    移动端设置rem单位

    非洲猫 8/15/2019 移动端

    在项目入口文件配置如下代码：

    setHtmlFontSize()
if (window.addEventListener) {
    window.addEventListener('resize', function () {
        setHtmlFontSize()
    }, false)
}
function setHtmlFontSize () {
    var deviceWidth;
    // 750是设计稿的宽度
    deviceWidth = document.documentElement.clientWidth;
    document.getElementsByTagName('html')[0].style.cssText = 'font-size:' + deviceWidth / 75.0 + 'px !important'
}

//index.css样式文件要设置如下样式：
html{font-size:10px;}/*设置rem单位,1rem为10px*/
